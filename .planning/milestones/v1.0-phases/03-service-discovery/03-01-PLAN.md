```typescript
import { Clinic, ClinicSearchParams, ApiResponse, Booking, Review } from '../types/clinic';

const API_BASE = '/api';

export async function searchClinics(params: ClinicSearchParams): Promise<ApiResponse<Clinic[]>> {
  // GET /api/clinics/search?lat=...&lng=...&radius=...&keyword=...
  const query = new URLSearchParams({
    lat: String(params.latitude),
    lng: String(params.longitude),
    radius: String(params.radius ?? 5000),
    ...(params.keyword && { keyword: params.keyword }),
  });
  
  const res = await fetch(`${API_BASE}/clinics/search?${query}`, {
    headers: { Authorization: `Bearer ${await getAuthToken()}` },
  });
  return res.json();
}

export async function getClinicDetails(clinicId: string): Promise<ApiResponse<Clinic>> {
  const res = await fetch(`${API_BASE}/clinics/${clinicId}`, {
    headers: { Authorization: `Bearer ${await getAuthToken()}` },
  });
  return res.json();
}

export async function getClinicReviews(clinicId: string): Promise<ApiResponse<Review[]>> {
  const res = await fetch(`${API_BASE}/clinics/${clinicId}/reviews`, {
    headers: { Authorization: `Bearer ${await getAuthToken()}` },
  });
  return res.json();
}
```

```typescript
// src/contexts/ClinicContext.tsx
import { createContext, useContext, useState, useCallback, ReactNode } from 'react';
import { Clinic } from '../types/clinic';
import { searchClinics, getClinicDetails } from '../services/clinic.service';

interface ClinicContextType {
  clinics: Clinic[];
  selectedClinic: Clinic | null;
  loading: boolean;
  error: string | null;
  searchClinicsNearby: (lat: number, lng: number, keyword?: string) => Promise<void>;
  selectClinic: (clinic: Clinic) => void;
  clearError: () => void;
}

const ClinicContext = createContext<ClinicContextType | undefined>(undefined);

export function ClinicProvider({ children }: { children: ReactNode }) {
  const [clinics, setClinics] = useState<Clinic[]>([]);
  const [selectedClinic, setSelectedClinic] = useState<Clinic | null>(null);
  const [loading, setLoading] = useState(false);
  const [error, setError] = useState<string | null>(null);

  const searchClinicsNearby = useCallback(async (lat: number, lng: number, keyword?: string) => {
    setLoading(true);
    setError(null);
    try {
      const result = await searchClinics({ latitude: lat, longitude: lng, keyword });
      if (result.success && result.data) {
        setClinics(result.data);
      } else {
        setError(result.error ?? 'Failed to search clinics');
      }
    } catch (e) {
      setError('Network error');
    } finally {
      setLoading(false);
    }
  }, []);

  const selectClinic = useCallback((clinic: Clinic) => {
    setSelectedClinic(clinic);
  }, []);

  const clearError = useCallback(() => setError(null), []);

  return (
    <ClinicContext.Provider value={{
      clinics, selectedClinic, loading, error,
      searchClinicsNearby, selectClinic, clearError,
    }}>
      {children}
    </ClinicContext.Provider>
  );
}

export function useClinic() {
  const context = useContext(ClinicContext);
  if (!context) throw new Error('useClinic must be used within ClinicProvider');
  return context;
}

export { ClinicContext };
```

**ClinicMap.tsx** - Map with markers (using WebView with Kakao Maps):
```typescript
// src/components/clinic/ClinicMap.tsx
import { View, StyleSheet } from 'react-native';
import { WebView } from 'react-native-webview';
import { Clinic } from '../../types/clinic';

interface Props {
  clinics: Clinic[];
  selectedClinic?: Clinic | null;
  onMarkerPress?: (clinic: Clinic) => void;
  latitude: number;
  longitude: number;
}

export function ClinicMap({ clinics, selectedClinic, onMarkerPress, latitude, longitude }: Props) {
  const mapHtml = `
    <!DOCTYPE html>
    <html>
    <head>
      <meta name="viewport" content="width=device-width, initial-scale=1.0">
      <script src="https://dapi.kakao.com/v2/maps/sdk.js?appkey=${KAKAO_MAP_KEY}"></script>
    </head>
    <body>
      <div id="map" style="width:100%;height:100%"></div>
      <script>
        const map = new kakao.maps.Map(document.getElementById('map'), {
          center: new kakao.maps.LatLng(${latitude}, ${longitude}),
          level: 4
        });
        
        const markers = [];
        ${clinics.map(c => `
          const marker${c.id} = new kakao.maps.Marker({
            position: new kakao.maps.LatLng(${c.latitude}, ${c.longitude}),
            title: '${c.name}'
          });
          marker${c.id}.setMap(map);
          kakao.maps.event.addListener(marker${c.id}, 'click', function() {
            window.ReactNativeWebView.postMessage(JSON.stringify({ type: 'marker', clinicId: '${c.id}' }));
          });
        `).join('')}
      </script>
    </body>
    </html>
  `;

  return (
    <View style={styles.container}>
      <WebView
        style={styles.map}
        source={{ html: mapHtml }}
        javaScriptEnabled
        onMessage={(event) => {
          try {
            const data = JSON.parse(event.nativeEvent.data);
            if (data.type === 'marker' && onMarkerPress) {
              const clinic = clinics.find(c => c.id === data.clinicId);
              if (clinic) onMarkerPress(clinic);
            }
          } catch {}
        }}
      />
    </View>
  );
}

const styles = StyleSheet.create({
  container: { height: 300, margin: 16, borderRadius: 12, overflow: 'hidden' },
  map: { flex: 1 },
});
```

```typescript
// OR app/clinic/index.tsx for expo-router
import { View, FlatList, StyleSheet, Text, ActivityIndicator } from 'react-native';
import { useEffect, useState } from 'react';
import * as Location from 'expo-location';
import { ClinicCard } from '../../components/clinic/ClinicCard';
import { ClinicMap } from '../../components/clinic/ClinicMap';
import { useClinic } from '../../contexts/ClinicContext';
import { Clinic } from '../../types/clinic';
import { router } from 'expo-router';

export default function ClinicSearchScreen() {
  const { clinics, loading, error, searchClinicsNearby } = useClinic();
  const [region, setRegion] = useState({ latitude: 37.5665, longitude: 126.9780 }); // Seoul default
  const [showMap, setShowMap] = useState(true);

  useEffect(() => {
    (async () => {
      const { status } = await Location.requestForegroundPermissionsAsync();
      if (status === 'granted') {
        const location = await Location.getCurrentPosition({});
        const { latitude, longitude } = location.coords;
        setRegion({ latitude, longitude });
        await searchClinicsNearby(latitude, longitude);
      }
    })();
  }, []);

  const handleClinicPress = (clinic: Clinic) => {
    router.push(`/clinic/${clinic.id}`);
  };

  if (loading && clinics.length === 0) {
    return (
      <View style={styles.centered}>
        <ActivityIndicator size="large" />
        <Text style={styles.loadingText}>Searching nearby clinics...</Text>
      </View>
    );
  }

  return (
    <View style={styles.container}>
      <ClinicMap
        clinics={clinics}
        selectedClinic={null}
        latitude={region.latitude}
        longitude={region.longitude}
        onMarkerPress={handleClinicPress}
      />
      <FlatList
        data={clinics}
        keyExtractor={(item) => item.id}
        renderItem={({ item }) => (
          <ClinicCard clinic={item} onPress={handleClinicPress} />
        )}
        contentContainerStyle={styles.list}
        ListEmptyComponent={
          <Text style={styles.emptyText}>No clinics found nearby</Text>
        }
      />
    </View>
  );
}

const styles = StyleSheet.create({
  container: { flex: 1, backgroundColor: '#f5f5f5' },
  centered: { flex: 1, justifyContent: 'center', alignItems: 'center' },
  loadingText: { marginTop: 12, fontSize: 16, color: '#666' },
```
